generate_synthetic.py
from datafoundry import Foundry # Initialize client foundry = Foundry(api_key="df_live_...") # Connect to source (stays in your VPC) source = foundry.sources.connect( type="snowflake", tables=["customers", "transactions"] ) # Generate 1M synthetic records synthetic = foundry.generate( source=source, volume=1_000_000, fidelity="high", privacy={"mode": "differential_privacy"} ) # Validate quality report = foundry.validate(synthetic) print(report.fidelity_score) # 0.98 print(report.privacy_score) # 1.0 (safe!) # Export for training synthetic.export("s3://bucket/data/")
